    Example

    import gfx.math.approx : approxUlp;

immutable ml = Mat!(float, 2, 3)(
    1, 2, 3,
    4, 5, 6,
);
immutable mr = Mat!(float, 3, 2)(
    1, 2,
    3, 4,
    5, 6,
);
immutable exp = Mat!(float, 2, 2)(
    1+6+15,  2+8+18,
    4+15+30, 8+20+36
);
assert(approxUlp(ml * mr, exp));
